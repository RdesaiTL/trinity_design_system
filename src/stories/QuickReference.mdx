## Basic Usage

```tsx
import { ThemeProvider } from '@mui/material/styles';
import { lightTheme, Button } from '@trinity/design-system';

function App() {
  return (
    <ThemeProvider theme={lightTheme}>
      <Button variant="contained">Get Started</Button>
    </ThemeProvider>
  );
}
```

## Styling Guidelines

### Colors

```tsx
// âœ… Use tokens
<Box sx={{ color: brandColors.primary.navy }} />
<Box sx={{ backgroundColor: semanticTokens.colors.background.primary }} />

// âŒ Never hardcode
<Box sx={{ color: '#0A2540' }} />  // VIOLATION
```

### Spacing & Radius

```tsx
// âœ… Use theme spacing
<Box sx={{ p: 2, gap: 1, borderRadius: 1 }} />

// âŒ Never hardcode
<Box sx={{ padding: '16px', borderRadius: '8px' }} />  // VIOLATION
```
